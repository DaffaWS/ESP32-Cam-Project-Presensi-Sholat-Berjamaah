// 🚀 Simple Utility Script

class ScriptDownloader {
    constructor(name) {
        this.name = name;
        this.version = '1.0.0';
        this.scripts = new Map();
    }

    async download(scriptName) {
        const script = this.scripts.get(scriptName);
        if (!script) throw new Error('Not found');
        return script;
    }

    addScript(name, code) {
        this.scripts.set(name, code);
        console.log(`✅ Added: ${name}`);
    }
}

const app = new ScriptDownloader('MyApp');
app.addScript('utils', 'console.log("Hello!")');
